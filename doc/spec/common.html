1. Introduction

The APIs described in this document are exposed in realsense-object-recognition, realsense_person_tracking and realsense_slam module.

2. Interface

2.1. Point2D

[
Constructor,

]
interface Point2D {
  attribute float x;
  attribute float y;

  void setCoords(sequence<float> coords);

2.3. Range

[
Constructor,

interface Range {
  attribute float begin;
  attribute float end;

  attribute float min; // Alias to begin
  attribute float max; // Alias to end

  readonly attribute float length;

  boolean contains(float value);
  boolean properlyContains(float value);

  boolean equal(Range range);

  boolean isEmpty();
  boolean isValid();

  stringifier;
};
